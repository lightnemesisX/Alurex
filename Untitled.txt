  <!-- Additional JavaScript for Interactive Features -->
  <script>
    // Fade-in effect for movie cards, testimonials, signup form, and features on page load
    document.addEventListener("DOMContentLoaded", function() {
      const fadeElements = document.querySelectorAll(".movie-card, .testimonial, .signup-form, .feature");
      fadeElements.forEach((el, index) => {
        el.style.opacity = 0;
        setTimeout(() => {
          el.style.transition = "opacity 1s";
          el.style.opacity = 1;
        }, index * 200);
      });
    });

    // Form validation before submission
    const signupForm = document.getElementById("signupForm");
    signupForm.addEventListener("submit", function(e) {
      let fullname = signupForm.fullname.value.trim();
      let email = signupForm.email.value.trim();
      let password = signupForm.password.value.trim();

      if (!fullname || !email || !password) {
        alert("Please fill in all the fields.");
        e.preventDefault();
      }
    });

    // Active navigation highlighting based on scroll position
    window.addEventListener("scroll", function() {
      const sections = document.querySelectorAll("section");
      const scrollPos = document.documentElement.scrollTop || document.body.scrollTop;

      sections.forEach(section => {
        if (scrollPos >= section.offsetTop - section.offsetHeight * 0.25 &&
            scrollPos < section.offsetTop + section.offsetHeight - section.offsetHeight * 0.25) {
          const id = section.getAttribute("id");
          document.querySelectorAll("nav ul li a").forEach(link => {
            link.classList.remove("active");
            if (link.getAttribute("href") === "#" + id) {
              link.classList.add("active");
            }
          });
        }
      });
    });
  </script>
